 * @param {number[]} putIn
 * @param {number[]} takeOut
 * @return {boolean}
 */
var validateBookSequences = function (putIn, takeOut) {
	let stack = [];
	let popIndex = 0;

	for (let item of putIn) {
		stack.push(item);
		while (stack.length > 0 && stack[stack.length - 1] == takeOut[popIndex]) {
			stack.pop();
			popIndex++;
		}
	}
	return stack.length == 0;
};
