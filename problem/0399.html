var calcEquation = function (equations, values, queries) {
	// æŠŠ equations æŠ½è±¡æˆä¸€å¹…å›¾ï¼Œé‚»æ¥è¡¨å­˜å‚¨
	let graph = new Map();
	for (let i = 0; i < equations.length; i++) {

			weight = values[i];

		// æ„å»ºåŒå‘å›¾

var bfs = function (graph, start, end) {
